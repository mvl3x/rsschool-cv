function findOdd(A) {
  const dict = {};

  for (let i = 0; i < A.length; i++) {
    (A[i] in dict) ? dict[A[i]] +=  1 : dict[A[i]] = 1;
  }

  for (key in dict) {

      return +key;
    }
  }
};
